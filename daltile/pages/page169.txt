                $('#headercontainer').removeClass('fixed');

   //The following function is used to fix the dropdown cutoff issue for Leed Tools category dropdown
            //The CSS width properties for the dropdown are declared in global.css
            //The jQuery selector in the function only selects the Leeds category dropdown
            function selectDropDownIEbehavior() {
                // Apply this behavior for IE only
                if (!($.browser.msie && $.browser.version == '7.0')) return;

                var expand = function () {
                        // Don't overwrite the stored original width,
                        // if the event occurs for a second time before contract()
                        if ($(this).data("origWidth")) return;
                        $(this).data("origWidth", $(this).outerWidth()).width("auto");
                };

                var contract = function () {
                        var origWidth = $(this).data("origWidth");
                        // If the original width was not stored, abort
                        if (!origWidth) return;
                        $(this).width(origWidth).removeData("origWidth");
                };

                $("#leedcatselect").each(function (index) {
                    // The select needs to be enclosed in a container with the same CSS width,
                    // which uses overflow:hidden, in order to hide the expanded part
                    var width = $(this).outerWidth();
                    var span = '<span style="width:' + width + '; overflow:hidden; float:left;"/>'; ;
                    $(this).wrap(span);

                    // Add event listeners
                    $(this).mousedown(expand).blur(contract).change(contract);
                });
            }
